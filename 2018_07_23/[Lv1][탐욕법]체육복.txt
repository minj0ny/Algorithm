class Solution {
	public int solution(int n, int[] lost, int[] reserve) {
		int answer = 0;
		answer = n;

		for (int i = 0; i < lost.length; i++) {
			boolean rent = false;
			for (int j = 0; j < reserve.length; j++) {
				int val = reserve[j];
				if (lost[i] == val || lost[i] == val + 1 || lost[i] == val - 1) {
					rent = true;
					reserve[j] = -1;
					break;
				}
			}
			if (!rent)
				answer--;
		}

		return answer;
	}
}
